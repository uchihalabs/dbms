# Function
```mysql
DELIMITER //
CREATE FUNCTION Calculate CafeMenu(cafeId INT)
RETURNS DECIMAL (10,2)
BEGIN
DECLARE menuItems Text;
SELECT GROUP_CONCAT(itemName Seperation) INTO menuItems
FROM menu
WHERE cafeId = cafeId;
RETURN menuItems;
END
```
